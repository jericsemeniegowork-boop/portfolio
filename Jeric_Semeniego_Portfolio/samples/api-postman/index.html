{
  "info": {
    "name": "Purchase Request QA Sample Collection",
    "schema": "https://schema.getpostman.com/json/collection/v2.1.0/collection.json",
    "description": "Representative API validation sample for a purchase request workflow."
  },
  "variable": [
    {
      "key": "base_url",
      "value": "https://example.test"
    },
    {
      "key": "request_id",
      "value": "PR-10025"
    }
  ],
  "item": [
    {
      "name": "Get purchase request details",
      "request": {
        "method": "GET",
        "header": [],
        "url": {
          "raw": "{{base_url}}/api/purchase-requests/{{request_id}}",
          "host": [
            "{{base_url}}"
          ],
          "path": [
            "api",
            "purchase-requests",
            "{{request_id}}"
          ]
        },
        "description": "Validates that the request is returned with an allowed status and matching ID."
      },
      "event": [
        {
          "listen": "test",
          "script": {
            "type": "text/javascript",
            "exec": [
              "pm.test(\"Status code is 200\", function () {",
              "  pm.response.to.have.status(200);",
              "});",
              "",
              "const data = pm.response.json();",
              "",
              "pm.test(\"Request ID matches\", function () {",
              "  pm.expect(data.requestId).to.eql(pm.variables.get('request_id'));",
              "});",
              "",
              "pm.test(\"Status is valid\", function () {",
              "  pm.expect(['Draft','Pending Approval','Approved','Rejected']).to.include(data.status);",
              "});"
            ]
          }
        }
      ]
    }
  ]
}
